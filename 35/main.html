Likelihood Gaussian discrepancy setup

%%Gaussian discrepancy error percentage related to observation in Bayesian inversion
scale_discrepancy{1} = 1; %(range 0.01~1)
scale_discrepancy{2} = 1; %(range 0.01~1)
 
%% Construct Gaussian discrepancy sigma^2 cell
for jj = 1:AnParam.N_outputfields
for kk = 1:AnParam.N_stage
sigma2{jj}(kk) = mean(observationsarray{jj}(kk),"all").*scale_discrepancy{jj};
end
end

Part 2: EoD + Bayesian inference

2.2 Recursive inference

for jj = 1:AnParam.N_stage
 
 
% enrich EoD manually and extract enriched FE restults
if jj==1 %First stage EoD run
[~] = EnrichedFERun(myUQLinkModel,ED, AnParam,filename,directory,jj);
else
[~] = EnrichedFERun(myUQLinkModel,ED, AnParam,filename,directory,jj, Posterior);
end
 
% Merge different EoD results to EDarray/FEcell_output if possible
% Update the number of FE runs if possible
[coord, EDarray, FEcell_output,AnParam] = merge_EoD(AnParam);
 
%% Construct PCA-PCE for a given stage {jj-th stage}
[myPCE, PCA] = PCEPCA(AnParam, EDarray, FEcell_output, jj, true);
 
%% Make PriorOpts/myPriorDist structure for parameter/discrepancy priors
%% Initial stage prior should be defined manually
 
if jj ==1
%intial srage prior should be defined manually
PriorOpts = []; Posterior{jj} = []; %initial empty cells
[PriorOpts, myPriorDist] = Dynamic_PriorOpts(ED,jj,AnParam, ...
observationsarray,sigma2,Posterior);
else
%rest stages should be inferred automatically
PriorOpts = []; %clear PriorOpts from last stage
[PriorOpts, myPriorDist] = Dynamic_PriorOpts(ED,jj,AnParam, ...
observationsarray,sigma2,Posterior);
end
 
%% === Observations ====
%%UQLAB doesnot support NAN in the myData.y
%%Repalce NAN with numerical value eplision=1e-20
epsilon = 1e-20;
for kk = 1:AnParam.N_outputfields
if ~isnan(observationsarray{kk}(:,jj))
obsconcat{kk} = observationsarray{kk}(:,jj)';
else
obsconcat{kk} = epsilon*ones(size(observationsarray{kk}(:,jj)))';
end
end
myData.y = obsconcat;
myData.Name = 'Deflection1 and Deflection2';
% %% === Likelihood switch ====
myLogLikeli = likeliSwitch(jj,observationsarray,PCA,myPCE,AnParam);
 
%% Solver options: AIES MCMC
Solver.Type = 'MCMC';
%Solver.MCMC.Visualize.Parameters = 1:AnParam.N_parameters;
%Solver.MCMC.Visualize.Interval = 20;
Solver.MCMC.Sampler = 'AIES';
Solver.MCMC.Steps = 100;
Solver.MCMC.NChains = 30;
BayesOpts.Data = myData;
BayesOpts.Type = 'inversion';
BayesOpts.LogLikelihood = myLogLikeli;
BayesOpts.Solver = Solver;
BayesOpts.Prior = myPriorDist;
BayesAnalysis = uq_createAnalysis(BayesOpts);
 
 
%% Postprocess (70% burn in; extract samples from posterior )
Posterior = BI_postprocess(jj,BayesAnalysis,AnParam,Posterior,gtarray,myPriorDist)
 
end
